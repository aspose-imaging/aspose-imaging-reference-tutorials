## คู่มือการใช้งาน

### รวมภาพหลายภาพเข้าเป็นภาพเดียว

ฟีเจอร์นี้แสดงให้เห็นวิธีการผสานรูปภาพสองรูปเป็นไฟล์เดียวโดยใช้ Aspose.Imaging สำหรับ .NET

#### กระบวนการทีละขั้นตอน

**1. กำหนดค่าตัวเลือก JPEG**

เริ่มต้นโดยการตั้งค่า `JpegOptions` ซึ่งจะกำหนดรูปแบบเอาต์พุตและคุณสมบัติของภาพรวมของคุณ

```csharp
using System;
using Aspose.Imaging;
using Aspose.Imaging.ImageOptions;

string dataDir = "YOUR_DOCUMENT_DIRECTORY";
string outputDir = "YOUR_OUTPUT_DIRECTORY";

// กำหนดค่า JpegOptions
JpegOptions imageOptions = new JpegOptions();
imageOptions.Source = new FileCreateSource(outputDir + "/CombinedImage_out.bmp", false);
```

**2. สร้างภาพใหม่**

เริ่มต้นใหม่ `Image` วัตถุที่มีขนาดตามต้องการโดยจะรวมภาพทั้งสองเข้าด้วยกัน

```csharp
using (var image = Image.Create(imageOptions, 600, 600))
{
    var graphics = new Graphics(image);

    // ล้างผ้าใบให้เป็นสีขาวก่อนวาดภาพ
    graphics.Clear(Color.White);
```

**3. วาดภาพ**

ใช้ `Graphics` วัตถุที่จะวางภาพของคุณลงบนผืนผ้าใบ

```csharp
    // วาดภาพแรกที่ครึ่งบนของผ้าใบ
    graphics.DrawImage(Image.Load(dataDir + "/sample_1.bmp"), 0, 0, 600, 300);

    // วาดภาพที่สองโดยตรงใต้ภาพแรก
    graphics.DrawImage(Image.Load(dataDir + "/File1.bmp"), 0, 300, 600, 300);
```

**4. บันทึกภาพรวม**

สุดท้ายให้บันทึกภาพรวมของคุณลงในดิสก์

```csharp
    // บันทึกผลลัพธ์เป็นไฟล์ใหม่
    image.Save();
}
```

### กำหนดค่าตัวเลือกภาพ

ความเข้าใจวิธีการกำหนดค่า `JpegOptions` เป็นสิ่งสำคัญสำหรับการเพิ่มประสิทธิภาพคุณภาพเอาต์พุตและการตั้งค่าเฉพาะรูปแบบ

#### การกำหนดค่า JPEGOptions

คุณสามารถตั้งค่าตัวเลือกเพิ่มเติมที่เหมาะกับความต้องการของคุณได้ดังนี้:

```csharp
using System;
using Aspose.Imaging.ImageOptions;

string dataDir = "YOUR_DOCUMENT_DIRECTORY";
string outputDir = "YOUR_OUTPUT_DIRECTORY";

// เริ่มต้น JpegOptions สำหรับการประมวลผลเพิ่มเติม
JpegOptions jpegOptions = new JpegOptions();
jpegOptions.Source = new FileCreateSource(outputDir + "/ProcessedImage_out.jpg", false);

// สามารถตั้งค่าการกำหนดค่าเพิ่มเติมได้ที่นี่ เช่น การตั้งค่าคุณภาพ
```
